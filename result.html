import matplotlib.pyplot as plt
from collections import Counter

def plot_relative_frequency(text):
    """
    Calculates and plots the relative frequency of letters in a given text.

    Args:
        text (str): The input text to analyze.
    """
    # Convert the text to lowercase and filter out non-alphabetic characters
    cleaned_text = ''.join(filter(str.isalpha, text.lower()))

    if not cleaned_text:
        print("Error: No alphabetic characters found in the text.")
        return

    # Count the frequency of each character
    char_counts = Counter(cleaned_text)

    # Calculate the total number of characters
    total_chars = len(cleaned_text)

    # Calculate the relative frequency of each character
    relative_frequencies = {char: count / total_chars for char, count in char_counts.items()}

    # Sort the frequencies alphabetically for consistent plotting
    sorted_freqs = sorted(relative_frequencies.items())
    chars, freqs = zip(*sorted_freqs)

    # Create the bar chart
    plt.figure(figsize=(12, 6))
    plt.bar(chars, freqs, color='skyblue')

    # Add labels and a title
    plt.title('Relative Frequency of Characters in a Text')
    plt.xlabel('Characters')
    plt.ylabel('Relative Frequency')
    plt.grid(axis='y', linestyle='--', alpha=0.7)

    # Display the plot
    plt.show()

# --- Example Usage ---

# You can replace this with your own text or a passage from a book.
sample_text = """
The quick brown fox jumps over the lazy dog.
This is an example text to demonstrate character frequency analysis.
Cryptanalysis often begins with frequency analysis to find patterns.
"""

# Call the function with the sample text
plot_relative_frequency(sample_text)

            # Avoid division by zero for expected_freq
            if expected_freq > 0:
                chi_squared += ((observed_freq - expected_freq) ** 2) / expected_freq
            # If expected_freq is 0 but observed is not, this contributes significantly
            elif observed_freq > 0:
                 chi_squared += observed_freq ** 2 / 0.01 # Add a small penalty, avoid inf

        # Update best shift if the current Chi-squared is lower
        if chi_squared < min_chi_squared:
            min_chi_squared = chi_squared
            best_shift = shift

    estimated_shifts[substring_name] = best_shift
    print(f"  {substring_name}: Best shift is {best_shift} (Chi-squared: {min_chi_squared:.2f})")


# Construct the potential key from the estimated shifts
# The shift found by the Chi-squared test is the amount we added to the ciphertext
# to make its frequency distribution look like English plaintext.
# To decrypt, we subtract the key. If C - K = P, then K = C - P.
# If shifting C by 's' gives P (C + s = P mod 26), then s = (P - C) mod 26.
# The shift 's' found here is the amount to ADD to the ciphertext character to get plaintext.
# The key character 'K' is the one such that C - K = P.
# So, K = (C - P) mod 26.
# If C + s = P, then s = P - C.
# K = -(P - C) = C - P.
# So K = -s mod 26, or (26 - s) mod 26.

potential_key_chars = [alphabet[(26 - shift) % 26] for shift in estimated_shifts.values()]
potential_key = "".join(potential_key_chars)

print(f"\nPotential key based on Chi-squared test: {potential_key}")

def vigenere_decrypt(encrypted_text, key):
    decrypted_text = ""
    alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    key = key.upper()
    key_length = len(key)

    for i, char in enumerate(encrypted_text.upper()):
        if char in alphabet:
            char_index = alphabet.index(char)
            key_char = key[i % key_length]
            key_index = alphabet.index(key_char)

            # Vigenere decryption formula: (C - K) mod 26
            decrypted_index = (char_index - key_index) % 26
            decrypted_char = alphabet[decrypted_index]
            decrypted_text += decrypted_char
        else:
            decrypted_text += char  # Handle non-alphabetic characters

    return decrypted_text

# Define the plot_relative_frequency function again for this cell's scope
def plot_relative_frequency(text, ax, title, color='skyblue'):
    """
    Calculates and plots the relative frequency of letters in a given text on a specified axes.

    Args:
        text (str): The input text to analyze.
        ax (matplotlib.axes.Axes): The axes object to plot on.
        title (str): The title for the plot.
        color (str): The color of the bars.
    """
    cleaned_text = ''.join(filter(str.isalpha, text.lower()))
    if not cleaned_text:
        print(f"Error: No alphabetic characters found in the text for {title}.")
        return

    char_counts = Counter(cleaned_text)
    total_chars = len(cleaned_text)
    relative_frequencies = {char: count / total_chars for char, count in char_counts.items()}

    sorted_freqs = sorted(relative_frequencies.items())
    chars, freqs = zip(*sorted_freqs)

    ax.bar(chars, freqs, color=color)
    ax.set_title(title)
    ax.set_xlabel('Characters')
    ax.set_ylabel('Relative Frequency')
    ax.grid(axis='y', linestyle='--', alpha=0.7)
